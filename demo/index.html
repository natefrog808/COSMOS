Workflow Patterns
// Example of form automation with COSMOS
const cosmos = await COSMOS.initialize({
  workspaceName: 'Form Automation Example'
});

// Start the agent
const agent = cosmos.start();

// Capture and analyze screen
await agent.run('takeScreenshot');
await agent.run('analyzeScreen');

// Find form fields
const nameField = await agent.run('findElement', {
  type: 'text_field',
  text: 'Name'
});

// Fill out form
await agent.run('interactWithElement', {
  elementId: nameField.elements[0].id,
  action: 'click'
});

await agent.run('typeText', {
  text: 'John Doe'
});
// Example of system monitoring with COSMOS
const cosmos = await COSMOS.initialize();

// Get detailed system information
const sysInfo = await agent.run('getSystemInfo');
console.log(`CPU: ${sysInfo.cpu.model} with ${sysInfo.cpu.cores} cores`);

// Monitor active window
setInterval(async () => {
  const windowInfo = await agent.run('detectActiveWindow');
  console.log(`Active window: ${windowInfo.title}`);
  
  // Send notification if memory usage exceeds threshold
  if (sysInfo.memory.usagePercent > 90) {
    await agent.run('sendNotification', {
      title: 'High Memory Usage',
      message: 'Memory usage above 90%',
      type: 'warning'
    });
  }
}, 5000);
// Example of recording and running workflow patterns
const cosmos = await COSMOS.initialize();
const agent = cosmos.start();

// Record a login pattern
await agent.run('takeScreenshot');
await agent.run('findElement', { text: 'Username' });
await agent.run('typeText', { text: 'admin' });
await agent.run('pressKey', { key: 'tab' });
await agent.run('typeText', { text: 'password123' });
await agent.run('findElement', { text: 'Login' });
await agent.run('clickMouse');

// Save this pattern for reuse
const pattern = await agent.run('recordPattern', {
  name: 'LoginSequence',
  description: 'Automated login to the system'
});

// Run the pattern later with different credentials
await agent.run('runPattern', {
  patternId: pattern.patternId,
  parameters: {
    username: 'different_user',
    password: 'secure_password'
  }
});
